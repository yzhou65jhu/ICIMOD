# This document lists the file structure of the operational and forecast system running on ICIMOD server.

$ROOTDIR = /home/kshakya

$ROOTDIR/
	|- data1/
	    |- SCRIPTS/					# scripts to download GDAS / run retrospective runs / forecast runs / LVT post-processing	
            |- Retro_NoahMP_5km/ 			# scripts to download CHIRPS/CHIRP / run LIS NoahMP with CHIRPS/CHIRP
	    |- MET_FORECING/                 		# all forcing data 
	        |- GDAS/                     		# all GDAS data
		|- CHIRP/				# all CHIRP forcing
		|- CHIRPS/				# all CHIRPS forcing
	    |- LISRUN					# LIS NoahMP3.6 CHIRPS/CHIRP + GDAS simulations
		|- CHIRP_diurnal/6HCHIRP 		# LDT CHIRP diurnal disaggregation
                |- CHIRPS_diurnal/CHIRPS6H		# LDT CHIRPS diurnal disaggregation
		|- Retro_NoahMP_GDAS_Chirp/NoahMP/SURFACEMODEL          # LIS NoahMP3.6 CHIRP+GDAS simulation results
		|- Retro_NoahMP_GDAS_Chirps/NoahMP/SURFACEMODEL         # LIS NoahMP3.6 CHIRPS+GDAS simulation results
	    |- GDAS_5KM_Noah3.6/					# retrospective run for Noah3.6 at 5km spatial resolution
		|- Noah3.6_GDAS_5km_output/		# retrospective run output directory for Noah3.6
	    |- GDAS_5KM_Tair/                    	# 5km maximum and minmum air temperature simulation
	        |- template.sa.gdas.5km/          
	    	    |- lis.config.temp.5km      	# LIS configure for maximum and minimum air temperature
                |- SURFACEMODEL/             	 	# 5km maximum and minimum air temperature results			   
	    |- FORECAST/				# running CLSM at 10km spatial resolution
		|- CLSM_Retro/				# retrospective run for CLSM at 10km spatial resolution
		    |- configs/                     	# LIS configure files for CLSM retrospective
		        |- lis.config_CLSM          	# LIS configure for operational CLSM retrospective run
			|- output/
	         	    |- OL_big_v2/
                               |- SURFACEMODEL/    	# retrospective run output directory for CLSM					
		|- SAForecast_CLSM/                 	# 10km SA forecast with CLSM directory
		    |- CLSM_EN_YYYYMM/              	# forecast result initailized at date YYYYMM01
			    |- SURFACEMODEL/            # forecast surface model result initailized at date YYYYMM01
			|- RSTFILE/                     # A directory for both retrospective restart files and ensemble restart files
			    |- LIS_RST_CLSMF25_YYYYMMDD2330.d01.nc     	# restart file copied from retrospective run
				|- LIS_enRST_YYYYMMDD2330.d01.nc       	# ensemble restart file for forecast (generated by LDT)
			|- lis.config.ensemble          # LIS configure for forecast run
				|- ldt.config.ensemble   	       	# LDT configure to generate ensemble restart file
		|- LVT_DI                     	 	# generate percentile drought indicators for CLSM retrospective and forecast output
		    |- configs
			|- lvt.config_forecast_perc_sf     		# config file of percentile for surface soil moisture
			|- lvt.config_forecast_perc_rtz    		# config file of percentile for rootzone soil moisture
			|- lvt.config_forecast_perc_tws    		# config file percentile for terrestrial water storage
		    |- output
			|- STATS_SF_bigSA    		# output of percentile for surface soil moisture
			|- STATS_RTZ_bigSA   		# output of percentile for rootzone soil moisture
			|- STATS_TWS_bigSA  	 	# outtput of percentile for terrestrial water storage

	|- zData/
		|- GDAS_5KM_RUN/					#output backup directory
			|- Noah3.6_GDASmet_pre_10_final/		#backup output for Noah3.6 at 5km spatial resolution
			|- CLSM_GDASmet_10_retro/			#backup output for CLSM at 10km spatial resolution
				|- CLSM_retro_final/			#output of retrospective run
				|- CLSM_forecast_final/			#output of forecast run
